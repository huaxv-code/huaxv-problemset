# 矩阵快速幂

=== "矩阵幂代码参考"

    ```java
	long[][] mqpow(long[][] a, int n, long x, long mod) {
		if (x == 0) {
			long[][] res = new long[n + 1][n + 1];
			
			for (int i = 1; i <= n; i ++) {
				res[i][i] = 1L;
			}
			
			return res;
		}
		
		if (x == 1L) return a;
		
		long[][] res = mqpow(a, n, x >> 1, mod);
		res = mmul(res, res, n, n, n, mod);
		if ((x & 1L) == 1L) res = mmul(res, a, n, n, n, mod);
		return res;
	}
    ```
