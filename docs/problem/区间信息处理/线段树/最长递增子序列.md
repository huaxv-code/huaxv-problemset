    import java.io.* ;
    import java.math.* ;
    import java.util.* ;

    class Solution {
        int[] as;
        Integer[] st;
        int ct, n, res;
        int[] tr;
        int[] rs;

        int lc(int p) { return p << 1; }
        int rc(int p) { return p << 1 | 1; }

        int unique(Integer nums[], int l, int r) {
            int i = l, j = l + 1;
            while (j < r) {
                if (nums[i] != nums[j]) {
                    nums[++ i] = nums[j];
                }
                j ++;
            }
            return i;
        }

        int lf(Integer nums[], int l, int r, int x) {
            int mid;
            while (l < r) {
                mid = (l + r) >> 1;
                if (nums[mid] < x) l = mid + 1;
                else r = mid;
            }
            if (nums[r] == x) return r;
            return -1;
        }

        void pushUp(int p, int l, int r) {
            if (l == r) return;
            tr[p] = Math.max(tr[lc(p)], tr[rc(p)]);
        }

        void build(int p, int l, int r) {
            if (l == r) {
                tr[p] = 0;
                return;
            }

            int mid = (l + r) >> 1;
            build(lc(p), l, mid);
            build(rc(p), mid + 1, r);

            pushUp(p, l, r);
        }

        void update(int p, int l, int r, int pos, int x) {
            if (pos <= l && r <= pos) {
                tr[p] = Math.max(tr[p], x);
                return;
            }

            int mid = (l + r) >> 1;

            if (pos <= mid) {
                update(lc(p), l, mid, pos, x);
            }
            if (pos > mid) {
                update(rc(p), mid + 1, r, pos, x);
            }

            pushUp(p, l, r);
            return;
        }

        int query(int p, int l, int r, int ll, int rr) {
            if (ll <= l && r <= rr) {
                return tr[p];
            }

            int a = 0, b = 0;
            int mid = (l + r) >> 1;

            if (ll <= mid) {
                a = query(lc(p), l, mid, ll, rr);
            }
            if (rr > mid) {
                b = query(rc(p), mid + 1, r, ll, rr);
            }

            return Math.max(a, b);
        }


        public int LIS(int[] nums) {
            ct = n = nums.length;

            as = new int[n + 1];
            st = new Integer[n + 1];
            tr = new int[(n + 1) << 2];
            rs = new int[n + 1];

            for (int i = 1; i <= n; i ++) {
                as[i] = nums[i - 1];
                st[i] = as[i];
            }

            Arrays.sort(st, 1, n + 1);  
            ct = unique(st, 1, n + 1);

            build(1, 1, ct);

            for (int i = 1; i <= n; i ++) {
                int t = lf(st, 1, ct, as[i]);
                if (t == ct) {
                    rs[i] = 1;
                } else {
                    rs[i] = query(1, 1, ct, t, ct) + 1;
                }
                update(1, 1, ct, t, rs[i]);
            }

            res = 1;

            for (int i = 1; i <= n; i ++) {
                res = Math.max(res, rs[i]);
            }

            return res;
        }
    }

    public class Main {

        static BufferedReader cin = 
            new BufferedReader (
                new InputStreamReader (System.in)
            );

        static PrintWriter cout = new PrintWriter (
            new OutputStreamWriter (System.out)
        );
        
        
        
        public void solve() throws Exception {
            
            Solution res = new Solution();

            String[] ts = cin.readLine().split(" ");
            int[] nums = new int[ts.length];
            for (int i = 0; i < ts.length; i ++) {
                nums[i] = Integer.parseInt(ts[i]);
            }

            cout.println(res.LIS(nums));
            cout.println(0);
            
            cout.flush();
        }
        
        public static void main(String[] args) throws Exception {
            
            Main cmd = new Main();
            cmd.solve();
            
        }
    }
