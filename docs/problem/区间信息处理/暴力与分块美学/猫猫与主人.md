# [猫猫与主人](https://ac.nowcoder.com/acm/problem/249989)

=== "暴力与分块美学"

    ```c++
    #include <bits/stdc++.h>

    using namespace std;
    typedef long long LL;
    typedef __int128_t HH;

    const LL N = 200000 + 200;
    const LL inf = ~(1ll << 63);

    struct Node {
        LL x, y;
        bool operator<(const Node& nd) const 
        {
            if (x == nd.x) return y < nd.y;
            return x < nd.x;
        } 
    };

    Node a[N], b[N];

    LL n, m;

    LL pos[N], st[N], ed[N], mi[N];
    LL len, cnt;

    LL find(LL x) {
        LL l = 1, r = m, mid;
        while (l < r) 
        {
            mid = (l + r) >> 1;
            if (b[mid].x < x) l = mid + 1;
            else r = mid;
        }
        if (b[r].x >= x) return r;
        return -1;
    }

    LL get_max(LL x, LL y) 
    {
        LL l = find(x), r = m;
        if (l == -1) return -1;
        LL p = pos[l], q = pos[r];
        if (p == q) 
        {
            for (LL i = r; i >= l; i --) 
            {
                if (b[i].y <= y) return b[i].x;
            }
            return -1;
        }
        else
        {
            for (LL i = r; i >= st[q]; i --) 
            {
                if (b[i].y <= y) return b[i].x;
            }
            for (LL i = q - 1; i >= p + 1; i --) 
            {
                if (mi[i] > y) continue;
                for (LL j = ed[i]; j >= st[i]; j --) 
                {
                    if (b[j].y <= y) return b[j].x;
                }
            }
            for (LL i = ed[p]; i >= l; i --) {
                if (b[i].y <= y) return b[i].x;
            }
            return -1;
        }
    }

    void solve()
    {
        
        scanf ("%lld%lld", &n, &m);

        for (LL i = 1; i <= n; i ++) scanf ("%lld", &a[i].x);
        for (LL i = 1; i <= n; i ++) scanf ("%lld", &a[i].y);

        for (LL i = 1; i <= m; i ++) scanf ("%lld", &b[i].x);
        for (LL i = 1; i <= m; i ++) scanf ("%lld", &b[i].y);

        sort(b + 1, b + 1 + m);

        len = sqrt(m);
        cnt = m / len;
        if (len * cnt < m) cnt ++;

        for (LL i = 1; i <= cnt; i ++) 
        {
            st[i] = (i - 1) * len + 1;
            ed[i] = i * len;
        }
        ed[cnt] = m;

        for (LL i = 1; i <= m; i ++) 
        {
            pos[i] = (i - 1) / len + 1;
        }

        for (LL i = 1; i <= cnt; i ++) 
        {
            mi[i] = inf;
            for (LL j = st[i]; j <= ed[i]; j ++) 
            {
                mi[i] = min(mi[i], b[j].y);
            }
        }

        for (LL i = 1; i <= n; i ++) 
        {
            printf ("%lld ", get_max(a[i].y, a[i].x));
        }
        puts("");
        
    }

    int main()
    {
        
        solve();
        
        return 0;
    }
    ```
