# 长度最小的子数组

??? success "代码参考"

    ```java
    class Solution {
        long[] st;
        int n;
        long[] as;

        int fd(int i, int j, long x) {
            if (i > j) return 0;
            int mid;

            while (i < j) {
                mid = (i + j + 1) >> 1;
                if (st[mid] <= x) i = mid;
                else j = mid - 1;
            }

            if (st[i] <= x) return i;
            return 0;
        }

        public int minSubArrayLen(int target, int[] nums) {
            n = nums.length;

            as = new long[n + 10];
            st = new long[n + 10];

            for (int i = 0; i < st.length; i ++) {
                st[i] = 0L;
            }

            for (int i = 1; i <= n; i ++) {
                as[i] = (long)nums[i - 1];
                st[i] = st[i - 1] + (long)as[i];
            }

            // Arrays.sort(st, 1, n + 1, (long x, long y) -> {
            //     return (int)(x - y);
            // }); // 常量不能自定义排序，报错

            Arrays.sort(st, 1, n + 1, (Long x, Long y) -> {
                return (int)(x - y);
            }); // 返回值必须是 int，一定要注意强制转换.

            // Arrays.sort(st, 1, n + 1, new Comparator<Long>(){
            //     public int compare(Long x, Long y) {
            //         return (int)(x - y);
            //     }
            // }); // 返回值必须是 int，一定要注意强制转换.

            int res = n + 1;
            for (int i = 1; i <= n; i ++) {
                if (st[i] < target) continue;
                res = Math.min(res, i);
                res = Math.min(res, i - fd(1, i - 1, st[i] - target));
            }

            if (res != n + 1) return res;
            return 0;
        }
    }
    ```
