# [教主的魔法](https://www.luogu.com.cn/problem/P2801)

=== "分块"

    ```c++
    #include <bits/stdc++.h>

    using namespace std;
    typedef long long LL;
    template <typename T> void read(T&);
    template <typename T> void print(T x);
    int gii(); LL gll(); double gff(); long double ghf();
    typedef __int128_t HH;
    const int N = 2000000;

    LL n, m;
    LL org[N], datas[N], tag[N], st[N], ed[N], pos[N];

    LL len, cnt;

    void add(LL l, LL r, LL x) {
        LL p = pos[l];
        LL q = pos[r];

        if (p == q) {
            for (LL i = l; i <= r; i ++) {
                org[i] += x;
            }

            for (LL i = st[pos[l]]; i <= ed[pos[l]]; i ++) {
                datas[i] = org[i];
            }

            sort(datas + st[pos[l]], datas + ed[pos[l]] + 1);
        }
        else {
            for (LL i = p + 1; i <= q - 1; i ++) {
                tag[i] += x;
            }
            for (LL i = l; i <= ed[pos[l]]; i ++) {
                org[i] += x;
            }
            for (LL i = st[pos[r]]; i <= r; i ++) {
                org[i] += x;
            }
            for (LL i = st[pos[l]]; i <= ed[pos[l]]; i ++) {
                datas[i] = org[i];
            }
            for (LL i = st[pos[r]]; i <= ed[pos[r]]; i ++) {
                datas[i] = org[i];
            }
            sort(datas + st[pos[l]], datas + ed[pos[l]] + 1);
            sort(datas + st[pos[r]], datas + ed[pos[r]] + 1);
        }
    }

    LL find(LL l, LL r, LL x) {
        LL mid;
        while (l < r) {
            mid = (l + r) >> 1;
            if (datas[mid] + tag[pos[mid]] < x) l = mid + 1;
            else r = mid;
        }
        if (datas[r] + tag[pos[r]] >= x) return r;
        return -1;
    }

    LL countx(LL l, LL r, LL x) {
        LL p = pos[l];
        LL q = pos[r];
        // cout << "p, q = " << p << ", " << q << endl;
        LL sum = 0;
        if (p == q) {
            for (LL i = l; i <= r; i ++) {
                if (org[i] + tag[pos[i]] >= x) sum ++;
            }
        } 
        else {
            for (LL i = p + 1; i <= q - 1; i ++) {
                LL t = find(st[i], ed[i], x);
                if (t == -1) continue;
                sum += ed[i] - t + 1;
                // cout << "t = " << ed[i] - t + 1 << endl;
            }
            for (LL i = l; i <= ed[pos[l]]; i ++) {
                if (org[i] + tag[pos[i]] >= x) sum ++;
            }
            for (LL i = st[pos[r]]; i <= r; i ++) {
                if (org[i] + tag[pos[i]] >= x) sum ++;
            }
        }
        return sum;
    }

    void solve() {
        
        n = gll();
        m = gll();

        for (LL i = 1; i <= n; i ++) {
            org[i] = gll();
            datas[i] = org[i];
        }

        while (len * len < n) len ++;
        if (len * len > n) len --;
        cnt = n / len;
        if (cnt * len < n) cnt ++;

        for (LL i = 1; i <= cnt; i ++) {
            st[i] = (i - 1) * len + 1;
            ed[i] = i * len;
        }

        ed[cnt] = n;

        for (LL i = 1; i <= n; i ++) {
            pos[i] = (i - 1) / len + 1;
        }

        for (LL i = 1; i <= cnt; i ++) {
            sort(datas + st[i], datas + ed[i] + 1);
        }

        char s[100];
        LL a, b, c;
        
        while (m --) {
            scanf ("%s%lld%lld%lld", s, &a, &b, &c);
            if (s[0] == 'M') {
                add(a, b, c);
            }
            else {
                // cout << "len = " << len << endl;
                // cout << "cnt = " << cnt << endl;
                printf ("%lld\n", countx(a, b, c));
            }
        }

    }

    int main() {

        solve();

        return 0;
    }

    int gii() {
        int x; scanf ("%d", &x); return x;
    }

    LL gll() {
        LL x; scanf ("%lld", &x); return x;
    }

    double gff() {
        double x; scanf ("%lf", &x); return x;
    }

    long double ghf() {
        long double x; scanf ("%Lf", &x); return x;
    }

    template <typename T> 
    void read(T& x) {
        x = 0;
        T f = 1;
        char c = getchar();
        while (!(c >= '0' && c <= '9')) {
            if (c == '-') f = -1;
            c = getchar();
        }
        while (c >= '0' && c <= '9') {
            x = x * 10 + c - '0';
            c = getchar();
        }
        x *= f;
    }

    template <typename T>
    void print(T x) {
        if (x == 0) {
            putchar('0'); return;
        }
        if (x < 0) {
            putchar('-'); x *= -1;
        }
        if (x / 10 != 0) print(x / 10);
        putchar('0' + (x % 10));
    }
    ```
