# [有向图的拓扑序列](https://www.acwing.com/problem/content/850/)

=== "队列 版的拓扑排序"

    ```java
    import java.util.* ;
    import java.io.* ;
    import java.math.* ;

    public class Main {
        
        int N = 1000000;
        int n, m;
        int[] du = new int[N];
        int[] g = new int[N];
        int[] e = new int[N];
        int[] ne = new int[N];
        int[] to = new int[N];
        int cnt, cur;

        java.util.Queue<Integer> qu = new ArrayDeque<>();

        void add(int x, int y) {
            cur ++;
            e[cur] = y;
            ne[cur] = g[x];
            g[x] = cur;
            du[y] ++;
        }
        
        public void solve() throws Exception {
            
            String[] ts = cin.readLine().split(space);
            n = gii(ts[0]); m = gii(ts[1]);

            for (int i = 1; i <= m; i ++) {
                ts = cin.readLine().split(space);
                int x, y;
                x = gii(ts[0]);
                y = gii(ts[1]);
                add(x, y);
            }

            for (int i = 1; i <= n; i ++) {
                if (du[i] == 0) {
                    qu.offer(i);
                }
            }

            while (qu.size() > 0) {
                int a = qu.poll();
                to[++ cnt] = a;
                for (int b = g[a]; b > 0; b = ne[b]) {
                    du[e[b]] --;
                    if (du[e[b]] == 0) {
                        qu.offer(e[b]);
                    } else if (du[e[b]] < 0) {
                        cout.println("-1"); return;
                    }
                }
            }

            if (cnt < n) {
                cout.println(-1); return;
            }

            for (int i = 1; i <= cnt; i ++) {
                cout.print(to[i] + " ");
            }

            cout.println();
            
        }

        public static void main(String[] args) throws Exception {
            
            Main cmd = new Main();
            cmd.solve();
            cout.flush();
            
        }

        public static String space = "\\s+"; 

        public static BufferedReader cin = new BufferedReader(new InputStreamReader(System.in));
        public static PrintWriter cout = new PrintWriter(new OutputStreamWriter(System.out));

        public static int gii(String s) {
            return Integer.parseInt(s);
        }

        public static long gll(String s) {
            return Long.parseLong(s);
        }

        public static double gff(String s) {
            return Double.parseDouble(s);
        }
    }
    ```

=== "DFS 版的拓扑排序"

    假设它的儿子都已经排好序，那么当前节点应该排在哪呢？

    ```java
    import java.util.* ;
    import java.io.* ;
    import java.math.* ;

    public class Main {
        
        int N = 1000000;
        int n, m;
        int[] g = new int[N];
        int[] e = new int[N];
        int[] ne = new int[N];
        int[] to = new int[N];
        int[] st = new int[N];
        int cnt, cur;

        // 假设它的儿子都已经排好序
        // 那么当前该节点应该排在哪呢？
        boolean dfs(int p) {
            if (p == 0) return true;
            if (st[p] == -1) { // 无法拓扑排序
                return false;
            }
            if (st[p] == 1) return true;
            st[p] = -1;

            for (int a = g[p]; a > 0; a = ne[a]) {
                if (dfs(e[a]) == false) return false;
            }

            to[++ cnt] = p;
            st[p] = 1;
            return true;
        }

        void add(int x, int y) {
            cur ++;
            e[cur] = y;
            ne[cur] = g[x];
            g[x] = cur;
        }
        
        public void solve() throws Exception {
            
            String[] ts = cin.readLine().split(space);
            n = gii(ts[0]); m = gii(ts[1]);

            for (int i = 1; i <= m; i ++) {
                ts = cin.readLine().split(space);
                int x, y;
                x = gii(ts[0]);
                y = gii(ts[1]);
                add(x, y);
            }

            for (int i = 1; i <= n; i ++) {
                dfs(i);
            }

            if (cnt < n) {
                cout.println(-1); return;
            }

            for (int i = cnt; i > 0; i --) {
                cout.print(to[i] + " ");
            }

            cout.println();
            
        }

        public static void main(String[] args) throws Exception {
            
            Main cmd = new Main();
            cmd.solve();
            cout.flush();
            
        }

        public static String space = "\\s+"; 

        public static BufferedReader cin = new BufferedReader(new InputStreamReader(System.in));
        public static PrintWriter cout = new PrintWriter(new OutputStreamWriter(System.out));

        public static int gii(String s) {
            return Integer.parseInt(s);
        }

        public static long gll(String s) {
            return Long.parseLong(s);
        }

        public static double gff(String s) {
            return Double.parseDouble(s);
        }
    }
    ```
