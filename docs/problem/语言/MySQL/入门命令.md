## updata 语句

```sql
update class_1
set age = 16
where name like "a%";
    -- 将名字以 a 开头的所有同学年龄改成 0 代表未知

update SC
set Grade = Grade + 5
where Cno = 3;
    -- 将编号 Cno = 3 的成绩都加 5

update c1_score
set subject = null,
    score = null
where name like "a%";
    -- 设置多列，以 a 开头的学生的课程及成绩都设置成未知


update c1_score t
inner join
(select name, subject, score from student_score) as sc
on t.name = sc.name and t.subject = sc.subject
set t.subject = sc.subject,
    t.score = sc.score;
    -- 用 select 语句从其他表中拿出数据插入
    -- 仅供 select 语句思维启蒙


update c1_score t
inner join
student_score as sc
on t.name = sc.name and t.subject = sc.subject
set t.subject = sc.subject,
    t.score = sc.score;
    -- 和上述一样的效果

update scores sc
inner join
(
    select name, sum(score) sm
    from scores
    group by name
) nsm
on sc.name = nsm.name
set sc.sum = nsm.sm;
    -- 仅供思维启蒙
```

```sql
表中的每个行都必须具有唯一主键，如果主键定的是某单一列，

*/

create table if not exists
scores
(
    id int not null auto_increment,
    name varchar(20) primary key,
    age int check(age between 16 and 24),
    subject varchar(20) not null check(subject in ("数学", "英语", "语文")),
```

## alter table 语句

```sql
alter table scores
add sum int check(sum between 0 and 300);
    -- 给 scores 表新增一列代表总分


update scores sc
inner join
(
    select name, sum(score) sm
    from scores
    group by name
) nsm
on sc.name = nsm.name
set sc.sum = nsm.sm;


-- -----------------------------------

alter table scores
drop column sum;
    -- 删除 sum 这一列

-- -----------------------------------

alter table scores
add constraint fk_subject -- 外键的名字
foreign key(subject)
references subject(name);
    -- 添加外键约束
    -- 外键必须参考其他表的主键
    -- 外键允许重复
    -- 外键约束，是为了添加新值时的范围必须限制在参考表的范围内
    -- 为什么要给外键取名？方便删除


alter table scores
drop foreign key fk_subject;

```

## 视图

```sql
/*
认识事物的最好方式：理解事物有什么用

那么视图有什么用呢？
*/

select c1.id, c1.name, c2.ct
    from (
        select * from class_1
            union all
        select * from class_1
    ) as c1 inner join (
        select id, count(id) as ct from (
            select * from class_1
                union all
            select * from class_1
        ) 
        as t 
        group by id
    ) as c2
    on c1.id = c2.id;

/*
还记得这一大坨查询语句吗？

select * from class_1
    union all 
select * from class_1

被重复使用了两次，可不可将
这个查询语句的内容包装成一
个虚拟表呢？

为什么叫做虚拟表？首先确实
含有类似表的数据，所以它是
表，但是它是通过查询语句检
索出来的数据，并不具体存储
，所以叫做虚拟的表。
*/

create view double_c1 as
select * from class_1
    union all
select * from class_1;

select c1.id, c1.name, c2.ct
from 
    double_c1 as c1 
    inner join
    (
        select id, count(id) as ct 
        from double_c1 as t
        group by id
    ) as c2
    on c1.id = c2.id;

-- 视图能更好的重用查询语句，
-- 将查询语句定义成一个视图（变量）的形式查询


drop view double_c1; 
    -- 删除视图

show tables; 
    -- 查看所有视图

create or replace view double_c1 
as
select * from class_1
    union all
select * from class_1;
    -- 覆盖视图
```

## 存储过程

```sql
-- 我个人觉得，存储过程就是个函数

show procedure status;
    -- 查看所有的存储过程

show procedure status like "avg%";
    -- 查看以 avg 开头的存储过程有哪些

delimiter // -- 设置结尾符为 //

create procedure avg_age()
begin
    select avg(age) as ag
    from class_1;
end //

delimiter ;  -- 设置结尾符为 ;

    -- 创建存储过程（函数），功能是显示 class_1 的平均年龄

drop procedure avg_age;
    -- 删除存储过程

-- -------------------------------------

delimiter //

create procedure fun
(
    out avg_age double,
    out cnt int,
    out old int,
    out young int
)
begin
    select avg(age) 
    into avg_age
    from class_1;

    select count(id) 
    into cnt
    from class_1;

    select max(age) 
    into old
    from class_1;

    select min(age)
    into young
    from class_1;
end //

delimiter ;

call fun
(
    @ag, @cnt, @old, @young
);

select @ag, @cnt, @old, @young;

-- 这个函数可以返回班级 class_1 的平均年龄、人数、最大年龄、最小年龄
-- 并将这些结果定义到变量 ag、cnt、old、young 中
-- 这是 out avg_age int 的作用，输出到变量的意思
-- 还有 in id int 是接受值是传入

-- -------------------------------------



```
