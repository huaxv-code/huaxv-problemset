    ```sql
    create table huaxv_table (
        h_name varchar(30) primary key, -- 不允许重复，且不允许为空，即：主键
        h_age int null, -- 默认就是允许重复和空
        h_sex int not null, -- 不允许为空
        h_year int,
        h_month int check (month >= 1 and moth <= 12), -- 设置取值范围为: 1 ~ 12
        h_day int check (day >= 1 and day <= 31),
        h_number varchar(11) unique check (h_number like "211250120")
    )
    ```

    create table huaxv_table (
