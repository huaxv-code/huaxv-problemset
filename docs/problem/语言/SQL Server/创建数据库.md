# 创建数据库

=== "sql"

    ```sql
    create database huaxv on primary (
        name = N'huaxv', -- 数据库名字
        filename = N'c:\data\huaxv.mdf', -- 数据库路径
        size = 10MB,     -- 初始大小
        maxsize = 200MB, -- 最大容量
        filegrowth = 20% -- 文件增长率
    )
    log on (
        name = N'huaxv_log',
        filename = N'huaxv_log.ldf',
        size = 5MB,
        maxsize = 100MB,
        filegrowth = 10%
    )
    go; -- 使用该数据库
    ```
