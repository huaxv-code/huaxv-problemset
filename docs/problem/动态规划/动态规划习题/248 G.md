# [248 G](https://www.luogu.com.cn/problem/P3146)

=== "线性 dp"

    ```c++
    #include <bits/stdc++.h>

    using namespace std;
    typedef long long LL;
    typedef __int128_t HH;
    const LL N = 500;

    LL read()
    {
        LL x = 0, f = 1;
        char c = getchar();
        while (!(c >= '0' && c <= '9'))
        {if (c == '-') f = -1; c = getchar();}
        while ((c >= '0' && c <= '9'))
        {x = x * 10 + c - '0'; c = getchar();}
        return x * f;
    }

    LL n;
    LL a[N];
    LL dp_max[N][N];
    LL dp_val[N][N];
    LL ans;

    void solve()
    {
        
        n = read();
        for (LL i = 1; i <= n; i ++) a[i] = read();
        for (LL i = 1; i <= n; i ++) 
        {
            dp_max[i][i] = dp_val[i][i] = a[i];
            ans = max(ans, a[i]);
        }
        
        for (LL i = 2; i <= n; i ++)
        {
            for (LL j = 1; j + i - 1 <= n; j ++)
            {
                for (LL k = j; k + 1 <= j + i - 1; k ++)
                {
                    if (dp_val[j][k] == dp_val[k + 1][j + i - 1])
                    {
                        if (dp_max[j][j + i - 1] < dp_val[j][k] + 1)
                        {
                            dp_max[j][j + i - 1] = dp_val[j][k] + 1;
                            dp_val[j][j + i - 1] = dp_val[j][k] + 1;
                            ans = max(ans, dp_max[j][j + i - 1]);
                        }
                    }
                }
            }
        }

        printf ("%lld\n", ans);
        
    }

    int main()
    {
        
        solve();
        
        return 0;
    }
    ```
